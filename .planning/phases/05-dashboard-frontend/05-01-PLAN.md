```python
class RetailerUrlResponse(BaseModel):
    id: int
    url: str
    created_at: datetime

class LatestScrapeResult(BaseModel):
    product_name: str
    price_cents: int
    listing_url: str
    scraped_at: datetime
    direction: str  # "new" | "higher" | "lower" | "unchanged"
    delta_cents: int
    pct_change: float

class RetailerUrlWithLatest(BaseModel):
    id: int
    url: str
    created_at: datetime
    latest_result: LatestScrapeResult | None = None

class WatchQueryResponse(BaseModel):
    id: int
    name: str
    threshold_cents: int
    is_active: bool
    schedule: str
    retailer_urls: list[RetailerUrlResponse]
    created_at: datetime
    updated_at: datetime

class WatchQueryDetailResponse(BaseModel):
    id: int
    name: str
    threshold_cents: int
    is_active: bool
    schedule: str
    retailer_urls: list[RetailerUrlWithLatest]
    created_at: datetime
    updated_at: datetime

class WatchQueryCreate(BaseModel):
    name: str
    threshold_cents: int
    urls: list[str]
    schedule: str = "daily"

class WatchQueryUpdate(BaseModel):
    name: str | None = None
    threshold_cents: int | None = None
    is_active: bool | None = None
    schedule: str | None = None
    urls: list[str] | None = None
```

From backend/app/schemas/alert.py:
```python
class AlertResponse(BaseModel):
    id: int
    watch_query_id: int
    watch_query_name: str
    product_name: str
    price_cents: int
    retailer_name: str
    listing_url: str
    is_read: bool
    created_at: datetime

class AlertSSEPayload(BaseModel):
    alert_id: int
    watch_query_id: int
    watch_query_name: str
    product_name: str
    price_cents: int
    retailer_name: str
    listing_url: str
    created_at: datetime
    unread_count: int

class UnreadCountResponse(BaseModel):
    unread_count: int
```

Backend API endpoints (from Phase 3-4 CONTEXT):
- GET /watch-queries/ -> WatchQueryResponse[]
- GET /watch-queries/{id} -> WatchQueryDetailResponse
- POST /watch-queries/ -> WatchQueryResponse (body: WatchQueryCreate)
- PATCH /watch-queries/{id} -> WatchQueryResponse (body: WatchQueryUpdate)
- DELETE /watch-queries/{id} -> 204
- POST /watch-queries/{id}/scrape -> ScrapeJobResponse
- GET /alerts/?limit=N -> AlertResponse[]
- GET /alerts/unread-count -> UnreadCountResponse
- PATCH /alerts/{id}/read -> AlertResponse
- POST /alerts/dismiss-all -> { dismissed_count: number }
- GET /alerts/stream -> SSE stream (event type: "alert", data: AlertSSEPayload JSON)

CORS: http://localhost:5173 and http://localhost:3000 allowed.
Pause/Resume: Use PATCH /watch-queries/{id} with { is_active: false/true }.
