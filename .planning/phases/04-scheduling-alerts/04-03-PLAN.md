```python
_sse_clients: set[asyncio.Queue] = set()

def add_sse_client(queue: asyncio.Queue) -> None:
    _sse_clients.add(queue)

def remove_sse_client(queue: asyncio.Queue) -> None:
    _sse_clients.discard(queue)

async def broadcast_alert(event_data: dict[str, Any]) -> None:
    for queue in _sse_clients.copy():
        try:
            queue.put_nowait(event_data)
        except asyncio.QueueFull:
            pass
```

    **1. Add SSE endpoint to `backend/app/api/alerts.py`:**

    This plan adds to the same file as Plan 02. Since Plans 02 and 03 are both Wave 2, the executor should coordinate -- but since this is a single endpoint addition, it can be appended.

    ```python
    import asyncio
    import json
    from fastapi import Request
    from fastapi.responses import StreamingResponse
    from app.services.alert_service import add_sse_client, remove_sse_client

    @router.get("/stream")
    async def alert_stream(request: Request):
        """SSE endpoint for real-time alert notifications."""
        queue: asyncio.Queue = asyncio.Queue(maxsize=50)
        add_sse_client(queue)

        async def event_generator():
            try:
                while True:
                    if await request.is_disconnected():
                        break
                    try:
                        data = await asyncio.wait_for(queue.get(), timeout=30.0)
                        # Serialize datetime fields to ISO format
                        json_data = json.dumps(data, default=str)
                        yield f"event: alert\ndata: {json_data}\n\n"
                    except asyncio.TimeoutError:
                        # Send keepalive comment to prevent connection timeout
                        yield ": keepalive\n\n"
            finally:
                remove_sse_client(queue)

        return StreamingResponse(
            event_generator(),
            media_type="text/event-stream",
            headers={
                "Cache-Control": "no-cache",
                "Connection": "keep-alive",
                "X-Accel-Buffering": "no",
            },
        )
    ```

    **IMPORTANT:** If `fastapi.sse.EventSourceResponse` is available (FastAPI >= 0.135.0), prefer using it instead of manual StreamingResponse. Try the import; if it works, use:
    ```python
    from fastapi.sse import EventSourceResponse, ServerSentEvent
    return EventSourceResponse(event_generator())
    ```
    where `event_generator` yields `ServerSentEvent(data=json_data, event="alert")` objects.
    If the import fails (version mismatch), use the manual StreamingResponse approach above.

    **IMPORTANT route ordering:** The `/stream` endpoint must be defined BEFORE `/{alert_id}/read` in the router, otherwise FastAPI will try to parse "stream" as an alert_id integer and return 422. Place the `@router.get("/stream")` endpoint before any path-parameter routes.

    ```python
    import asyncio
    from app.services.alert_service import broadcast_alert, _sse_clients

    async def test_sse_stream_receives_broadcast(client):
        """Test that SSE stream receives broadcast alert events."""
        # We need to test this by:
        # 1. Start an SSE connection in a background task
        # 2. Broadcast an alert event
        # 3. Verify the event arrives on the stream

        event_data = {
            "alert_id": 1,
            "watch_query_id": 1,
            "watch_query_name": "Test Query",
            "product_name": "Test Product",
            "price_cents": 999,
            "retailer_name": "Amazon",
            "listing_url": "https://amazon.com/test",
            "created_at": "2026-03-18T00:00:00",
            "unread_count": 5,
        }

        received_lines = []

        async def read_stream():
            async with client.stream("GET", "/alerts/stream") as response:
                assert response.status_code == 200
                async for line in response.aiter_lines():
                    received_lines.append(line)
                    if "alert_id" in line:
                        break  # Got our event, stop reading

        # Start stream reader as background task
        task = asyncio.create_task(read_stream())
        # Give the connection time to establish
        await asyncio.sleep(0.1)

        # Broadcast alert event
        await broadcast_alert(event_data)

        # Wait for stream to receive (with timeout)
        try:
            await asyncio.wait_for(task, timeout=5.0)
        except asyncio.TimeoutError:
            task.cancel()

        # Verify event was received
        data_lines = [l for l in received_lines if l.startswith("data:")]
        assert len(data_lines) >= 1
        assert "alert_id" in data_lines[0]
        assert "Test Product" in data_lines[0]

    async def test_sse_client_cleanup():
        """Test that disconnected clients are cleaned up."""
        from app.services.alert_service import _sse_clients, add_sse_client, remove_sse_client
        queue = asyncio.Queue(maxsize=50)
        add_sse_client(queue)
        assert queue in _sse_clients
        remove_sse_client(queue)
        assert queue not in _sse_clients
    ```

    Note: The SSE streaming test with httpx may require special handling since the test client uses ASGITransport. If `client.stream()` does not work with ASGITransport, test the SSE broadcast mechanism at the unit level instead:
    - Create a queue, add it as SSE client, broadcast, assert data arrived in queue.
    - This tests the core broadcast logic without needing a full HTTP SSE connection.
